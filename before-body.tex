\newcommand{\chbcolorbox}[5]{%
  \setlength{\fboxrule}{2pt}%                                 % Set the thickness of the frame box rule to 2 points
  \textcolor[HTML]{d95f02}{%                                  % Set the text color to green using its HTML hex code
    \fbox{%                                                   % Create a framed box around the content
      \begin{minipage}{0.35\textwidth}%                       % Create a minipage with a width of 35% of the text width
        \centering%                                           % Center the content within the minipage
        \textcolor[HTML]{1b9e77}{#1}\\%                       % Center the first argument and set its color to blue
        \textcolor[HTML]{1b9e77}{#2}\\%                       % Center the second argument and set its color to blue
        \textcolor[HTML]{1b9e77}{#3}\\%                       % Center the third argument and set its color to blue
        \textcolor[HTML]{1b9e77}{#4}\\%                       % Center the fourth argument and set its color to blue
        \textcolor[HTML]{1b9e77}{#5}%                         % Center the fourth argument and set its color to blue
      \end{minipage}%                                         % End the minipage environment
    }%
  }%
}
